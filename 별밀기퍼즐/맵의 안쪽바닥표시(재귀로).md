```text
def floodFill(mapObj, x, y, oldCharacter, newCharacter)
    # 플레이어시작점(x,y)도 바닥으로 셋팅
    if mapObj[x][y] == oldCharacter:  
	    mapObj[x][y] = newCharacter  

    # '#'(벽)을 만나면 멈춤
	if x < len(mapObj) - 1 and mapObj[x+1][y] == oldCharacter:  
	    floodFill(mapObj, x+1, y, oldCharacter, newCharacter) #오른쪽호출  
	if x > 0 and mapObj[x-1][y] == oldCharacter:  
	    floodFill(mapObj, x-1, y, oldCharacter, newCharacter) # 외쪽호출  
	if y < len(mapObj[x]) - 1 and mapObj[x][y+1] == oldCharacter:  
	    floodFill(mapObj, x, y+1, oldCharacter, newCharacter) #아래호출  
	if y > 0 and mapObj[x][y-1] == oldCharacter:  
	    floodFill(mapObj, x, y-1, oldCharacter, newCharacter) #위호출
```
