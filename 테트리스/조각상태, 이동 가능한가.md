def addToBoard(board, piece):  
    for x in range(TEMPLATEWIDTH):  
        for y in range(TEMPLATEHEIGHT):  
            if PIECES[piece['shape']][piece['rotation']][y][x]                  != BLANK:  
                   board[x + piece['x']][y + piece['y']] =                                                        piece['color']  
    # This happens after a piece has landed
```

for 구문안에 있는 x와 y는 조각의 조각내부인덱스값
`board[x + piece['x']]` 에서 x는 조각내부인덱스값 ,
`piece['x']` 박스인덱스값

def isValidPosition(board, piece, adjX=0, adjY=0):  
    # Return True if the piece is 
      within the board and not colliding  
    for x in range(TEMPLATEWIDTH):  
        for y in range(TEMPLATEHEIGHT):  
            isAboveBoard = y + piece['y'] + adjY < 0  
            if isAboveBoard or 
                PIECES[piece['shape']][piece['rotation']][y][x]                                                        == BLANK:  
                continue  
            if not isOnBoard(x + piece['x'] + adjX,
                             y + piece['y'] + adjY):  
                return False  
            if board[x + piece['x'] + adjX]
                    [y + piece['y'] + adjY] != BLANK:  
                return False  
    return True
```
